#### Response Body

<APIBlock>
  <APIField name="dryRun" type="Boolean">
    Whether the request was a dry run or the requested action was applied.
  </APIField>
  <APIField name="hardDelete" type="Boolean">
    Whether the requested action was deactivation or permanent deletion.
  </APIField>
  <APIField name="total" type="Integer">
    The count of affected Users.
  </APIField>
  <APIField name="userIds" type="Array<UUID>">

  </APIField>
