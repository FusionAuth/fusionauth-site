  <APIField name="tenantManagerConfiguration" type="Object">
    The Tenant Manager configuration object.
  </APIField>

  <APIField name="tenantManagerConfiguration.applicationConfigurations" type="Array<Object>">
    The list of universal applications that are enabled for Tenant Manager identity providers.
  </APIField>
  <APIField name="tenantManagerConfiguration.applicationConfigurations[x].applicationId" type="UUID">
    The Id of a universal application enabled for Tenant Manager identity providers.
  </APIField>

  <APIField name="tenantManagerConfiguration.attributeFormId" type="UUID" optional>
    The Id of the registration form used to determine available attribute mapping targets.
  </APIField>

    The brand name displayed in Tenant Manager.
  </APIField>

  <APIField name="tenantManagerConfiguration.identityProviderTypeConfigurations" type="Object">
    A map of tenant manager IdP type configurations keyed by Identity Provider type.
  </APIField>
  <APIField name="tenantManagerConfiguration.identityProviderTypeConfigurations[type].defaultAttributeMappings" type="Map<String, String>">
    Default attribute mappings for this allowed Identity Provider type.
  </APIField>
  <APIField name="tenantManagerConfiguration.identityProviderTypeConfigurations[type].enabled" type="Boolean">
    Whether this Identity Provider type is enabled for Tenant Manager.
  </APIField>
  <APIField name="tenantManagerConfiguration.identityProviderTypeConfigurations[type].insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that this Identity Provider type configuration was created.
  </APIField>
  <APIField name="tenantManagerConfiguration.identityProviderTypeConfigurations[type].lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that this Identity Provider type configuration was last updated.
  </APIField>
  <APIField name="tenantManagerConfiguration.identityProviderTypeConfigurations[type].linkingStrategy" type="String">
    The linking strategy used for Identity Providers of this type.
  </APIField>
  <APIField name="tenantManagerConfiguration.identityProviderTypeConfigurations[type].type" type="String">
    The identity provider type key for this configuration.
  </APIField>

  <APIField name="tenantManagerConfiguration.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Tenant Manager configuration was created.
  </APIField>
  <APIField name="tenantManagerConfiguration.lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Tenant Manager configuration was last updated.
  </APIField>
