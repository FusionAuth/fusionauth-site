#### Response Body

<APIBlock>
  <APIField name="scope.applicationId" type="UUID">
    The unique Id of the Application.
  </APIField>
  <APIField name="scope.data" type="Object">
    An object that can hold any information about the Scope that should be persisted.
  </APIField>
  <APIField name="scope.defaultConsentDetail" type="String">
    The default detail shown when prompting a user for consent.
  </APIField>
  <APIField name="scope.defaultConsentMessage" type="String">
    The default message shown when prompting a user for consent.
  </APIField>
  <APIField name="scope.description" type="String">
    The description for the Scope.
  </APIField>
  <APIField name="scope.id" type="UUID">
    The unique Id of the Scope.
  </APIField>
  <APIField name="scope.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Scope was added to the FusionAuth database.
  </APIField>
  <APIField name="scope.lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Scope was updated in the FusionAuth database.
  </APIField>
  <APIField name="scope.name" type="String">
    The name of the Scope. This value will be used in OAuth workflows to request scopes and in tokens to indicate consented scopes.
  </APIField>
  <APIField name="scope.required" type="Boolean">
    Whether the scope is required. Users must consent to required scopes to complete the OAuth workflow.
  </APIField>
</APIBlock>
