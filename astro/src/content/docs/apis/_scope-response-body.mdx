<APIBlock>
  <APIField name="scope.applicationId" type="UUID">
    The unique Id of the Application.
  </APIField>

  </APIField>
  <APIField name="scope.defaultConsentDetail" type="String">
    The default detail to display on the OAuth consent screen if one cannot be found in the theme.
  </APIField>
  <APIField name="scope.defaultConsentMessage" type="String">
    The default message to display on the OAuth consent screen if one cannot be found in the theme.
  </APIField>
  <APIField name="scope.description" type="String">
    A description of the OAuth Scope for internal use.
  </APIField>
  <APIField name="scope.id" type="UUID">
    The unique Id of the OAuth Scope.
  </APIField>
  <APIField name="scope.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the OAuth Scope was added to the FusionAuth database.
  </APIField>
  <APIField name="scope.lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the OAuth Scope was updated in the FusionAuth database.
  </APIField>
  <APIField name="scope.name" type="String">
    The name of the OAuth Scope. This is the value that will be used to request the scope in OAuth workflows.
  </APIField>
  <APIField name="scope.required" type="Boolean">
    Determines if the OAuth Scope is required when requested in an OAuth workflow.
  </APIField>
