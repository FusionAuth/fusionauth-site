#### Response Body

<APIBlock>
  <APIField name="action.actioneeUserId" type="UUID">
    The Id of the User that was actioned.
  </APIField>
  <APIField name="action.actionerUserId" type="UUID">
    The Id of the User that took the action on the User. If the action was initiated by FusionAuth this value will not be provided.
  </APIField>
  <APIField name="action.applicationIds" type="Array<UUID>">
    This parameter if provided specifies the scope of the User Action. When an Action is scoped to one or more Applications the Application Ids will be provided in this parameter.
  </APIField>
  <APIField name="action.comment" type="String">
    An optional comment provided when the Action was created, updated or canceled. This value will always be the last comment set on the action, see `history` for previous values.
  </APIField>
  <APIField name="action.emailUserOnEnd" type="Boolean">
    Whether FusionAuth will email the User when a time-based User Actions expires.
  </APIField>
  <APIField name="action.endEventSent" type="Boolean">
    Whether FusionAuth will send events when a time-based User Actions expires.
  </APIField>
  <APIField name="action.expiry" type="Long">
    The expiration [instant](/docs/reference/data-types#instants) of the User Action.
  </APIField>
  <APIField name="action.history.historyItems" type="Array" renderif={props.userActionIncludeHistory}>
    The historical data for the User Action Log. Each time the User Action is modified or when the User Action is canceled a new historyItem is recorded.
  </APIField>
  <APIField name="action.history.historyItems[x].actionerUserId" type="UUID" renderif={props.userActionIncludeHistory}>
    The Id of the User that took the modified (or created) the User Action.
  </APIField>
  <APIField name="action.history.historyItems[x].comment" type="String" renderif={props.userActionIncludeHistory}>
    An optional comment provided when the Action was created, updated or canceled.
  </APIField>
  <APIField name="action.history.historyItems[x].createInstant" type="Long" renderif={props.userActionIncludeHistory}>
    The [instant](/docs/reference/data-types#instants) that this historical modification or creation was performed.
  </APIField>
  <APIField name="action.history.historyItems[x].expiry" type="Long" renderif={props.userActionIncludeHistory}>
    The [instant](/docs/reference/data-types#instants) that the User Action expired at previously.
  </APIField>
  <APIField name="action.id" type="UUID">
    The Id of the User Action record.
  </APIField>
  <APIField name="action.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the User Action was taken.
  </APIField>
  <APIField name="action.localizedOption" type="String">
    The localized version of the User Action Option that was selected by the actioner.
  </APIField>
  <APIField name="action.localizedReason" type="String">
    The localized version of the User Action Reason that was selected by the actioner.
  </APIField>
  <APIField name="action.notifyUserOnEnd" type="String">
    Whether or not FusionAuth will send events to registered Webhooks when a time-based User Action expires.
  </APIField>
  <APIField name="action.option" type="String">
    The non-localized version of the User Action Option that was selected by the actioner.
  </APIField>
  <APIField name="action.reason" type="String">
    The non-localized version of the User Action Reason that was selected by the actioner.
  </APIField>
  <APIField name="action.reasonCode" type="String">
    The User Action Reason code that was selected by the actioner.
  </APIField>
  <APIField name="action.userActionId" type="UUID">
    The unique Id of the User Action. This Id can be used to retrieve the User Action using the [Retrieve a User Action](/docs/apis/user-actions#retrieve-a-user-action) API.
  </APIField>
</APIBlock>
