#### Response Body

<APIBlock>
  <APIField name="groups" type="Array">
    The list of Group objects.
  </APIField>
  <APIField name="groups[x].data" type="Object">
    An object that can hold any information about the Group that should be persisted.
  </APIField>
  <APIField name="groups[x].id" type="UUID">
    The name of the Group.
  </APIField>
  <APIField name="group[x].insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Group was added to the FusionAuth database.
  </APIField>
  <APIField name="group[x].lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Group was updated in the FusionAuth database.
  </APIField>
  <APIField name="groups[x].name" type="String">
    The name of the Group.
  </APIField>
  <APIField name="groups[x].roles" type="Map<UUID, List<Object>">
    The roles assigned to this Group. The map key is the Application Id, the value is an array of Application Roles.
  </APIField>
  <APIField name="groups[x].roles[applicationId][x].description" type="String">
    The description of the role.
  </APIField>
  <APIField name="groups[x].roles[applicationId][x].id" type="UUID">
    The Application Role Unique Id
  </APIField>
  <APIField name="groups[x].roles[applicationId][x].name" type="String">
    The name of the Group.
  </APIField>
  <APIField name="groups[x].roles[applicationId][x].isDefault" type="Boolean">
    Whether or not the Role is a default role. A default role is automatically assigned to a user during registration if no roles are provided.
  </APIField>
  <APIField name="groups[x].roles[applicationId][x].isSuperRole" type="Boolean">
    Whether or not the Role is a considered to be a super user role. This is a marker to indicate that it supersedes all other roles. FusionAuth will attempt to enforce this contract when using the web UI, it is not enforced programmatically when using the API.
  </APIField>
  <APIField name="groups[x].tenantId" type="UUID">
    The unique Id of the Tenant.
  </APIField>
  <APIField name="total" type="Integer" renderif={!!props.include_total}>
    The total number of Groups matching the search criteria. Use this value along with the <InlineField>numberOfResults</InlineField> and <InlineField>startRow</InlineField> in the Search request to perform pagination.
  </APIField>
</APIBlock>
