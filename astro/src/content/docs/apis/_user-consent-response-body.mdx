#### Response Body

<APIBlock>
  <APIField name="userConsent.consent" type="Object">
    The Consent object represented by the <InlineField>consentId</InlineField> field.
  </APIField>
  <APIField name="userConsent.consentId" type="UUID">
    The Id of the Consent being granted to the User.
  </APIField>
  <APIField name="userConsent.giverUserId" type="UUID">
    The Id of the User giving consent. When providing self-consent the <InlineField>giverUserId</InlineField> will be the same as the <InlineField>userId</InlineField>.
  </APIField>
  <APIField name="userConsent.id" type="UUID">
    The unique Id of the User consent.
  </APIField>
  <APIField name="userConsent.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) when the User consent was granted.
  </APIField>
  <APIField name="userConsent.lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) when the User consent was last updated.
  </APIField>
  <APIField name="userConsent.status" type="String">
    The current status of the User consent. Possible values are:

    - `Active`
    - `Revoked`
  </APIField>
  <APIField name="userConsent.userId" type="UUID">
    The Id of the User receiving consent. When providing self-consent the <InlineField>giverUserId</InlineField> will be the same as the <InlineField>userId</InlineField>.
  </APIField>
  <APIField name="userConsent.values" type="Array<String>">
    The consent values to be associated with this User consent.
  </APIField>
</APIBlock>
