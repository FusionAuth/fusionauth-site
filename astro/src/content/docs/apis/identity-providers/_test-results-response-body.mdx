<h4> Response Body </h4>
<div class="border-b-2"></div>

<APIBlock>
  <APIField name="result" type="Object">
    The test result payload for the specified connection test Id.
  </APIField>
  <APIField name="result.success" type="Boolean">
    Whether the test completed successfully. This is `true` only when at least one step exists and all recorded steps succeeded.
  </APIField>
  <APIField name="result.startInstant" type="DateTime">
    The instant at which the test began.
  </APIField>
  <APIField name="result.identityProviderId" type="UUID">
    The Identity Provider Id used for the test.
  </APIField>
  <APIField name="result.identityProviderUserId" type="String" optional>
    The third-party user Id resolved during the test, if available.
  </APIField>
  <APIField name="result.email" type="String" optional>
    The reconciled email address, if available.
  </APIField>
  <APIField name="result.username" type="String" optional>
    The reconciled username, if available.
  </APIField>
  <APIField name="result.steps" type="Array<Object>">
    Ordered trace steps captured during the connection test.
  </APIField>
  <APIField name="result.steps[].title" type="String">
    The step title.
  </APIField>
  <APIField name="result.steps[].success" type="Boolean">
    Whether the step succeeded.
  </APIField>
  <APIField name="result.steps[].detail" type="String">
    Additional detail for the step.
  </APIField>
</APIBlock>
