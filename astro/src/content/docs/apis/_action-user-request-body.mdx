#### Request Body

<APIBlock>
  <APIField name="action.actioneeUserId" type="UUID" required>
    The Id of the User that is being actioned.
  </APIField>
  <APIField name="action.actionerUserId" type="UUID" required>
    The Id of the User that is taking the action on the User.
  </APIField>
  <APIField name="action.applicationIds" type="Array<UUID>" optional>
    The list of Application ids that the action is being performed in.
  </APIField>
  <APIField name="action.comment" type="String" optional>
    The comment left by the actioner.
  </APIField>
  <APIField name="action.emailUser" type="Boolean" optional defaults="false">
    Whether FusionAuth should send an email to the User.
  </APIField>
  <APIField name="action.expiry" type="Long" optional>
    The expiration [instant](/docs/reference/data-types#instants) of this User Action. This value is required for time-based User Actions.

    To cause the action to be applied indefinitely, or until the action is canceled or modified, set this value to `9223372036854775807`.
  </APIField>
  <APIField name="action.notifyUser" type="Boolean" optional defaults="false">
    The **notifyUser** flag that is passed along in any events FusionAuth sends to registered Webhooks.
  </APIField>
  <APIField name="action.option" type="String" optional>
    The User Action Option that the actioner selected.
  </APIField>
  <APIField name="action.reasonId" type="UUID" optional>
    The Id of the User Action Reason that the actioner selected.
  </APIField>
  <APIField name="action.userActionId" type="UUID" required>
    The Id of the User Action that the actioner is performing on the User.
  </APIField>
  <APIField name="broadcast" type="Boolean" optional defaults="false">
    Whether or not FusionAuth will broadcast the User Action to any registered Webhooks.
  </APIField>
</APIBlock>
