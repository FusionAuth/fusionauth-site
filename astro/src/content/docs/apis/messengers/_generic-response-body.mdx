#### Response Body

<APIBlock>
  <APIField name="messenger.connectTimeout" type="Integer">
    The connect timeout for the HTTP connection, in milliseconds.
  </APIField>
  <APIField name="messenger.data" type="Object">
    An object that can hold any information about the Messenger that should be persisted.
  </APIField>
  <APIField name="messenger.debug" type="Boolean">
    If debug is enabled, an event log is created to assist in debugging messenger errors.
  </APIField>
  <APIField name="messenger.headers" type="Object">
    An object that can hold HTTPHeader key and value pairs.
  </APIField>
  <APIField name="messenger.httpAuthenticationPassword" type="String">
    The basic authentication password to use for requests to the Messenger.
  </APIField>
  <APIField name="messenger.httpAuthenticationUsername" type="String">
    The basic authentication username to use for requests to the Messenger.
  </APIField>
  <APIField name="messenger.id" type="UUID">
    The unique Id of the Messenger.
  </APIField>
  <APIField name="messenger.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) when the Messenger was created.
  </APIField>
  <APIField name="messenger.lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) when the Messenger was last updated.
  </APIField>
  <APIField name="messenger.name" type="String">
    The unique Messenger name.
  </APIField>
  <APIField name="messenger.readTimeout" type="Integer">
    The read timeout for the HTTP connection, in milliseconds.
  </APIField>
  <APIField name="messenger.sslCertificateId" type="UUID">
    The Id of an existing [Key](/docs/apis/keys). The X.509 certificate is used for client certificate authentication in requests to the Messenger.
  </APIField>
  <APIField name="messenger.transport" type="String">
    The message transport protocol used to send a message. This value is currently read-only and set to `sms`.
  </APIField>
  <APIField name="messenger.type" type="String">
    This field will always be `Generic`.
  </APIField>
  <APIField name="messenger.url" type="String">
    The fully qualified URL of the endpoint that will accept requests from FusionAuth.
  </APIField>
</APIBlock>
