#### Response Body

<APIBlock>
  <APIField name="messenger.accountSID" type="String">
    The Twilio Account ID to use when connecting to the Twilio API.
  </APIField>
  <APIField name="messenger.authToken" type="String">
    The Twilio Auth Token to use when connecting to the Twilio API.
  </APIField>
  <APIField name="messenger.data" type="Object">
    An object that can hold any information about the Messenger that should be persisted.
  </APIField>
  <APIField name="messenger.debug" type="Boolean">
    If debug is enabled, an event log is created to assist in debugging messenger errors.
  </APIField>
  <APIField name="messenger.fromPhoneNumber" type="String">
    The configured Twilio phone number that will be used to send messages.
  </APIField>
  <APIField name="messenger.id" type="UUID">
    The unique Id of the Messenger.
  </APIField>
  <APIField name="messenger.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Messenger was added to the FusionAuth database.
  </APIField>
  <APIField name="messenger.lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Messenger was last updated in the FusionAuth database.
  </APIField>
  <APIField name="messenger.messagingServiceSid" type="String">
    The Twilio message service Id, this is used when using Twilio Copilot to load balance between numbers.

    When using <InlineField>messagingServicesSid</InlineField>, you may omit the <InlineField>fromPhoneNumber</InlineField> field.
  </APIField>
  <APIField name="messenger.name" type="String">
    The unique Messenger name.
  </APIField>
  <APIField name="messenger.transport" type="String">
    The message transport protocol used to send a message. This value is currently read-only and set to `sms`.
  </APIField>
  <APIField name="messenger.type" type="String">
    This field will always be `Twilio`.
  </APIField>
  <APIField name="messenger.url" type="String">
    The fully qualified URL of the Twilio API endpoint that will accept requests from FusionAuth.
  </APIField>
</APIBlock>
