#### Response Body

<APIBlock>
  <APIField name="typeConfiguration" type="Object">
    The Tenant Manager identity provider type configuration object.
  </APIField>

  <APIField name="typeConfiguration.defaultAttributeMappings" type="Map<String, String>">
    A map of default attribute mappings for this identity provider type.
  </APIField>
  <APIField name="typeConfiguration.defaultAttributeMappings[key]" type="String">
    The source expression used to populate the mapped FusionAuth user or registration field.
  </APIField>

  <APIField name="typeConfiguration.enabled" type="Boolean">
    Whether this identity provider type is enabled for Tenant Manager.
  </APIField>

  <APIField name="typeConfiguration.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that this identity provider type configuration was created.
  </APIField>

  <APIField name="typeConfiguration.lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that this identity provider type configuration was last updated.
  </APIField>

  <APIField name="typeConfiguration.linkingStrategy" type="String">
    The linking strategy for this identity provider type configuration.
  </APIField>

  <APIField name="typeConfiguration.type" type="String">
    The identity provider type key for this configuration.
  </APIField>
</APIBlock>
