#### Response Body

<APIBlock>
  <APIField name="actions" type="Array">
    The list of User Actions.
  </APIField>
  <APIField name="actions[x].actioneeUserId" type="UUID">
    The Id of the User that was actioned.
  </APIField>
  <APIField name="actions[x].actionerUserId" type="UUID">
    The Id of the User that was took the action on the User.
  </APIField>
  <APIField name="actions[x].applicationIds" type="Array<UUID>">
    The list of Application ids that the action was performed in.
  </APIField>
  <APIField name="actions[x].comment" type="String">
    The comment left by the actioner. This is the last User to touch the User Action (i.e. if the User Action was updated, this will be the comment left by the User that updated it).
  </APIField>
  <APIField name="actions[x].emailUserOnEnd" type="Boolean">
    Whether FusionAuth will email the User when a time-based User Actions expires.
  </APIField>
  <APIField name="actions[x].endEventSent" type="Boolean">
    Whether FusionAuth will send events when a time-based User Actions expires.
  </APIField>
  <APIField name="actions[x].expiry" type="Long">
    The expiration [instant](/docs/reference/data-types#instants) of the User Action.
  </APIField>
  <APIField name="actions[x].history.historyItems" type="Array">
    The historical data for the User Action. Each time the User Action is modified a new historyItem is added to the list.
  </APIField>
  <APIField name="actions[x].history.historyItems[x].actionerUserId" type="UUID">
    The Id of the User that took the modified (or created) the User Action.
  </APIField>
  <APIField name="actions[x].history.historyItems[x].comment" type="String">
    The message that the actioner added when they modified (or created) this User Action.
  </APIField>
  <APIField name="actions[x].history.historyItems[x].createInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that this historical modification or creation was performed.
  </APIField>
  <APIField name="actions[x].history.historyItems[x].expiry" type="Long">
    The [instant](/docs/reference/data-types#instants) that the User Action expired at previously.
  </APIField>
  <APIField name="actions[x].id" type="UUID">
    The Id of the User Action.
  </APIField>
  <APIField name="action[x].insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the User Action was taken.
  </APIField>
  <APIField name="actions[x].localizedOption" type="String">
    The localized version of the User Action Option that was selected by the actioner.
  </APIField>
  <APIField name="actions[x].localizedReason" type="String">
    The localized version of the User Action Reason that was selected by the actioner.
  </APIField>
  <APIField name="actions[x].notifyUserOnEnd" type="String">
    Whether or not FusionAuth will send events to registered Webhooks when a time-based User Action expires.
  </APIField>
  <APIField name="actions[x].option" type="String">
    The non-localized version of the User Action Option that was selected by the actioner.
  </APIField>
  <APIField name="actions[x].reason" type="String">
    The non-localized version of the User Action Reason that was selected by the actioner.
  </APIField>
  <APIField name="actions[x].reasonCode" type="String">
    The User Action Reason code that was selected by the actioner.
  </APIField>
</APIBlock>
