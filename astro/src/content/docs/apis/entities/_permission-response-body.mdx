#### Response Body

<APIBlock>
  <APIField name="permission.data" type="Object">
    An object that can hold any information about the Permission that should be persisted.
  </APIField>
  <APIField name="permission.description" type="String">
    The description of the Permission.
  </APIField>
  <APIField name="permission.id" type="UUID">
    The Id of the Permission.
  </APIField>
  <APIField name="permission.isDefault" type="Boolean">
    Whether or not the Permission is a default permission. A default permission is automatically granted to an entity of this type if no permissions are provided in a grant request.
  </APIField>
  <APIField name="permission.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Permission was added to the FusionAuth database.
  </APIField>
  <APIField name="permission.lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Permission was updated in the FusionAuth database.
  </APIField>
  <APIField name="permission.name" type="String">
    The name of the Permission. Once created, this field cannot be changed.
  </APIField>
</APIBlock>
