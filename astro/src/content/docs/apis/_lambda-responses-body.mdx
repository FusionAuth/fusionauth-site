import APIBlock from 'src/components/api/APIBlock.astro';
import APIField from 'src/components/api/APIField.astro';

#### Response Body

<APIBlock>
  <APIField name="lambdas" type="Array">
    The list of Lambda objects.
  </APIField>
  <APIField name="lambdas[x].body" type="String">
    The lambda function body, a JavaScript function.
  </APIField>
  <APIField name="lambdas[x].debug" type="Boolean">
    Whether or not debug event logging is enabled for this Lambda.
  </APIField>
  <APIField name="lambdas[x].enabled" type="Boolean" deprecated>
    Whether or not this Lambda is enabled.

    Not currently used and may be removed in a future version.
  </APIField>
  <APIField name="lambdas[x].engineType" type="String">
    The JavaScript execution engine for the lambda. The possible values are:

    * `GraalJS`
    * `Nashorn` 
  </APIField>
  <APIField name="lambdas[x].id" type="UUID">
    The Id of the Lambda.
  </APIField>
  <APIField name="lambdas[x].insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the lambda created.
  </APIField>
  <APIField name="lambdas[x].lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the lambda was last updated.
  </APIField>
  <APIField name="lambdas[x].name" type="String">
    The name of the lambda.
  </APIField>
</APIBlock>
