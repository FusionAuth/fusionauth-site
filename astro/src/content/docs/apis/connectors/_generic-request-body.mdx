#### Request Body

<APIBlock>
  <APIField name="connector.authenticationURL" type="String" required>
    The fully qualified URL used to send an HTTP request to authenticate the user. Must be a valid URL.
  </APIField>
  <APIField name="connector.connectTimeout" type="Integer" required>
    The connect timeout for the HTTP connection, in milliseconds. Value must be greater than `0`.
  </APIField>
  <APIField name="connector.data" type="Object" optional>
    An object that can hold any information about the Connector that should be persisted.
  </APIField>
  <APIField name="connector.debug" type="Boolean" optional defaults="false">
    Determines if debug should be enabled to create an event log to assist in debugging integration errors.
  </APIField>
  <APIField name="connector.headers" type="Object" optional>
    An object that can hold HTTPHeader key and value pairs.
  </APIField>
  <APIField name="connector.httpAuthenticationPassword" type="String" optional>
    The basic authentication password to use for requests to the Connector.
  </APIField>
  <APIField name="connector.httpAuthenticationUsername" type="String" optional>
    The basic authentication username to use for requests to the Connector.
  </APIField>
  <APIField name="connector.name" type="String" required>
    The unique Connector name.
  </APIField>
  <APIField name="connector.readTimeout" type="Integer" required>
    The read timeout for the HTTP connection, in milliseconds. Value must be greater than `0`.
  </APIField>
  <APIField name="connector.sslCertificateKeyId" type="UUID" optional>
    The Id of an existing [Key](/docs/apis/keys). The X.509 certificate is used for client certificate authentication in requests to the Connector.
  </APIField>
  <APIField name="connector.type" type="String" required>
    This field must be set to `Generic`.
  </APIField>
</APIBlock>
