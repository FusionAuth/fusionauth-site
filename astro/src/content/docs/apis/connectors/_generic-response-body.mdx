#### Response Body

<APIBlock>
  <APIField name="connector.authenticationURL" type="String">
    The fully qualified URL used to send an HTTP request to authenticate the user.
  </APIField>
  <APIField name="connector.connectTimeout" type="Integer">
    The connect timeout for the HTTP connection, in milliseconds. Value must be greater than `0`.
  </APIField>
  <APIField name="connector.data" type="Object">
    An object that can hold any information about the Connector that should be persisted.
  </APIField>
  <APIField name="connector.debug" type="Boolean">
    Determines if debug should be enabled to create an event log to assist in debugging integration errors.
  </APIField>
  <APIField name="connector.headers" type="Object">
    An object that can hold HTTPHeader key and value pairs.
  </APIField>
  <APIField name="connector.httpAuthenticationPassword" type="String">
    The basic authentication password to use for requests to the Connector.
  </APIField>
  <APIField name="connector.httpAuthenticationUsername" type="String">
    The basic authentication username to use for requests to the Connector.
  </APIField>
  <APIField name="connector.id" type="UUID">
    The unique Id of the Connector.
  </APIField>
  <APIField name="connector.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) when the Connector was created.
  </APIField>
  <APIField name="connector.lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) when the Connector was last updated.
  </APIField>
  <APIField name="connector.name" type="String">
    The unique Connector name.
  </APIField>
  <APIField name="connector.readTimeout" type="Integer">
    The read timeout for the HTTP connection, in milliseconds. Value must be greater than `0`.
  </APIField>
  <APIField name="connector.sslCertificateKeyId" type="UUID">
    The Id of an existing [Key](/docs/apis/keys). The X.509 certificate is used for client certificate authentication in requests to the Connector.
  </APIField>
  <APIField name="connector.type" type="String">
    This field must be set to `Generic`.
  </APIField>
</APIBlock>
