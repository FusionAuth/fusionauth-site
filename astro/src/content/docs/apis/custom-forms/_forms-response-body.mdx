#### Response Body

<APIBlock>
  <APIField name="forms" type="Array">
    The list of Form objects.
  </APIField>
  <APIField name="forms[x].data" type="Object">
    An object that can hold any information about the Form that should be persisted.
  </APIField>
  <APIField name="forms[x].id" type="UUID">
    The unique Id of the Form.
  </APIField>
  <APIField name="forms[x].insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Form was added to the FusionAuth database.
  </APIField>
  <APIField name="forms[x].lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Form was last updated in the FusionAuth database.
  </APIField>
  <APIField name="forms[x].name" type="String">
    The unique name of the Form.
  </APIField>
  <APIField name="forms[x].steps" type="Array<Object>">
    An ordered list of objects containing one or more Form Fields.
  </APIField>
  <APIField name="forms[x].steps[x].fields" type="Array<UUID>">
    An ordered list of Form Field Id's assigned to this step.
  </APIField>
  <APIField name="forms[x].type" type="String">
    The Form type. The possible values are:

     * <InlineFieldValue>adminRegistration</InlineFieldValue> - This form can be used to customize the add and edit User Registration form in the FusionAuth UI. <AvailableSince since="1.20.0" />
     * <InlineFieldValue>adminUser</InlineFieldValue> - This form can be used to customize the add and edit User form in the FusionAuth UI. <AvailableSince since="1.20.0" />
     * <InlineFieldValue>registration</InlineFieldValue> - This form can be used for self service registration.
    * <InlineFieldValue>selfServiceUser</InlineFieldValue> - This form can be used for Self Service Account Management. <AvailableSince since="1.26.0" />
  </APIField>
</APIBlock>
