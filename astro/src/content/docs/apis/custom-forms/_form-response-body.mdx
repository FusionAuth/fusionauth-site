#### Response Body

<APIBlock>
  <APIField name="form.data" type="Object">
    An object that can hold any information about the Form that should be persisted.
  </APIField>
  <APIField name="form.id" type="UUID">
    The unique Id of the Form.
  </APIField>
  <APIField name="form.insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Form was added to the FusionAuth database.
  </APIField>
  <APIField name="form.lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Form was last updated in the FusionAuth database.
  </APIField>
  <APIField name="form.name" type="String">
    The unique name of the Form.
  </APIField>
  <APIField name="form.steps">
    An ordered list of objects containing one or more Form Fields.
  </APIField>
  <APIField name="form.steps[x].fields">
    An ordered list of Form Field Ids assigned to this step.
  </APIField>
  <APIField name="form.steps[x].type" type="String" optional defaults="collectData" since="1.62.0">
    The type of step being created.
      * `collectData` - Standard form step type that collects fields from the user.
      * `verifyEmail` - Perform verification on <InlineField>user.email</InlineField> before user creation.
      * `verifyPhoneNumber` - Perform verification on <InlineField>user.phoneNumber</InlineField> before user creation.
  </APIField>
  <APIField name="form.type" type="String">
    The form type. The possible values are:

    * `adminRegistration` - This form can be used to customize the add and edit User Registration form in the FusionAuth UI. <AvailableSince since="1.20.0" />
     * `adminUser` - This form can be used to customize the add and edit User form in the FusionAuth UI. <AvailableSince since="1.20.0" />
     * `registration` - This form can be used for self service registration.
     * `selfServiceUser` - This form can be used for [Self Service Account Management](/docs/lifecycle/manage-users/account-management/). <AvailableSince since="1.26.0" />
  </APIField>
</APIBlock>
