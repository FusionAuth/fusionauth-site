#### Response Body

<APIBlock>
  <APIField name="userConsents[x].consent" type="Object">
    The Consent object represented by the **consentId** field.
  </APIField>
  <APIField name="userConsents[x].consentId" type="UUID">
    The Id of the Consent being granted to the User.
  </APIField>
  <APIField name="userConsents[x].giverUserId" type="UUID">
    The Id of the User giving consent. When providing self-consent the **giverUserId** will be the same as the **userId**.
  </APIField>
  <APIField name="userConsents[x].id" type="UUID">
    The unique Id of the User consent.
  </APIField>
  <APIField name="userConsents[x].insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) when the User consent was granted.
  </APIField>
  <APIField name="userConsents[x].lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) when the User consent was last updated.
  </APIField>
  <APIField name="userConsents[x].status" type="String">
    The current status of the User consent. Possible values are:

    - `Active`
    - `Revoked`
  </APIField>
  <APIField name="userConsents[x].userId" type="UUID">
    The Id of the User receiving consent. When providing self-consent the **giverUserId** will be the same as the **userId**.
  </APIField>
  <APIField name="userConsents[x].values" type="Array<String>">
    The consent values to be associated with this User consent.
  </APIField>
</APIBlock>
