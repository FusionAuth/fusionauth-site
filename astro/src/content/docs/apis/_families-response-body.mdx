#### Response Body

<APIBlock>
  <APIField name="families" type="Array<Object>">
    The list of Family objects.
  </APIField>
  <APIField name="families[x].id" type="UUID">
    The unique Id of the family.
  </APIField>
  <APIField name="families[x].insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Family was added to the FusionAuth database.
  </APIField>
  <APIField name="families[x].lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) that the Family was updated in the FusionAuth database.
  </APIField>
  <APIField name="families[x].members" type="Array<Object>">
    The members of this family.
  </APIField>
  <APIField name="families[x].members[x].insertInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) when the user was added to the family.
  </APIField>
  <APIField name="families[x].members[x].lastUpdateInstant" type="Long">
    The [instant](/docs/reference/data-types#instants) when the family member was last updated.
  </APIField>
  <APIField name="families[x].members[x].owner" type="Boolean">
    True if this user is the owner of the family, the first Adult user in the family will automatically be set as the owner. A teen or child cannot be the family owner.
  </APIField>
  <APIField name="families[x].members[x].role" type="String">
    The role of the family member. The possible values are:

    * `Adult`
    * `Child`
    * `Teen`
  </APIField>
  <APIField name="families[x].members[x].userId" type="UUID">
    The unique user Id of the family member.
  </APIField>
</APIBlock>
