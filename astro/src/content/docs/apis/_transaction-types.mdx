import InlineFieldValue from 'src/components/InlineFieldValue.astro';

* <InlineFieldValue>None</InlineFieldValue> - No Webhooks are required to succeed for the FusionAuth transaction to be committed.
* <InlineFieldValue>Any</InlineFieldValue> - Only a single Webhook is required to succeed for the FusionAuth transaction to be committed.
* <InlineFieldValue>SimpleMajority</InlineFieldValue> - A simple majority (50% or more) of Webhooks are required to succeed for the FusionAuth transaction to be committed.
* <InlineFieldValue>SuperMajority</InlineFieldValue> - A super majority (2/3 or more) of Webhooks are required to succeed for the FusionAuth transaction to be committed.
* <InlineFieldValue>AbsoluteMajority</InlineFieldValue> - Every Webhook must succeed for the FusionAuth transaction to be committed.
